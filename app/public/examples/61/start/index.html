Umieść następujący kod w pliku functions.php szablonu

function load_jquery() {
    if (!is_admin()) {
        wp_deregister_script('jquery');
        wp_register_script('jquery', '//ajax.googleapis.com/ajax/libs/jquery/2.2.1/jquery.min.js', false, '2.2.1', false);
        wp_enqueue_script('jquery');
    }
}

add_action('after_setup_theme', 'load_jquery');
